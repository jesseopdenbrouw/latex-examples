\documentclass[a4paper,12pt]{article}

\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{xcolor}

%% Indexing package
\usepackage{imakeidx}
% Add some text after the index title!
\indexprologue[\vspace*{36pt}]{This is the index.}
% Or you can apply another font size!
\indexsetup{
othercode=\footnotesize,
}

% Make the index and add a line to the TOC
\makeindex[intoc]

%% Three-column index
%\makeindex[intoc,columns=3]

\begin{document}

Some text. Note: upper case and lower case letter are different.
