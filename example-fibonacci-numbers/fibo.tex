\documentclass[a4paper]{article}


% The counters
\newcount\n \newcount\fibo \newcount\fiboprev \newcount\temp

% Print Fibonacci numbers using an iterative approach
\def\fibonacci#1{%
\n=#1 \advance\n by -2 \fibo=1 \fiboprev=0
0, 1%
\loop
	\temp=\fibo
	\advance\fibo by \fiboprev
	\fiboprev=\temp
	\ifnum\n>1\relax , \else{} and~\fi
	\the\fibo
	\advance\n by -1\relax
	\ifnum\n>0\relax
\repeat
}

\begin{document}
\raggedright

\noindent
The first 47 Fibonacci numbers are \fibonacci{47}.

\end{document}
